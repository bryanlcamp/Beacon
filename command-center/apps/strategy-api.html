Algorithm Container.

Plug in your strategies into Beacon's framework.
Simple API. HFT performance. Focus on what matters.

Example API Usage:

#pragma once
#include <beacon_algorithm>
#include <beacon_order>

class ExampleStrategy : public user_algorithms::BaseAlgorithm {
public:
    explicit ExampleStrategy(const nlohmann::json& config) : BaseAlgorithm(config) {}

    void onMarketDataReceived(const MarketMessage& msg) override {
        // Your trading logic here
    }

    void onOrderConfirmed(const Order& order) override {}
    void onOrderFiled(const OrderFill& fill) override {}
    void onOrderRejected(const OrderRequest& req) override {}
    void onExchangeMessageReceived(const ExchangeMessage& msg) override {}
};

REGISTER_ALGORITHM(ExampleStrategy);
